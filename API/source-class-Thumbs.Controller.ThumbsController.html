28: /**
29:  * Creates and displays thumbnails for image and video files
30:  */
31: class ThumbsController extends AppController {
32:     /**
33:      * Internal function to render a thumbnail
34:      * @param string $target Target file
35:      */
36:     protected function _render($target) {
37:         if(is_url($target))
38:             return $this->redirect($target);
39:         
40:         $this->autoRender = FALSE;
41:         
42:         //Renders the thumbnail
43:         header(sprintf('Content-type: %s', mime_content_type($target)));
44:         readfile($target);
45:         
46:         exit;
47:     }
48: 
49:     /**
50:      * Resizes an images, creating a thumbnail.
51:      * 
52:      * You have to set the maximum width and/or the maximum height as query string parameters (`width` and `height` parameters).
53:      * @param string $origin Origin file path, encoded with `urlencode()` and `base64_encode()`
54:      * @uses Thumbs\Utility\ThumbCreator::resize()
55:      * @uses _render()
56:      */
57:     public function resize($origin) {       
58:         $target = (new ThumbCreator(urldecode(base64_decode($origin))))
59:             ->resize($this->request->query('width'), $this->request->query('height'));
60:         
61:         return $this->_render($target);
62:     }
63:     
64:     /**
65:      * Resizes an images, creating a square thumbnail.
66:      * 
67:      * You have to set the maximum side as query string parameter (`side` parameter).
68:      * @param string $origin Origin file path, encoded with `urlencode()` and `base64_encode()`
69:      * @uses Thumbs\Utility\ThumbCreator::square()
70:      * @uses _render()
71:      */
72:     public function square($origin) {
73:         $target = (new ThumbCreator(urldecode(base64_decode($origin))))
74:             ->square($this->request->query('side'));
75:         
76:         return $this->_render($target);
77:     }
78:     
79:     /**
80:      * Convenient alias for `resize()` and `square()` actions.  
81:      * It determines which method to use depending on the query arguments.
82:      * @param string $origin Origin file path, encoded with `urlencode()` and `base64_encode()`
83:      * @uses resize()
84:      * @uses square()
85:      */
86:     public function thumb($origin) {
87:         if($this->request->query('side'))
88:             return $this->square($origin);
89:         else
90:             return $this->resize($origin);
91:     }
92: }
