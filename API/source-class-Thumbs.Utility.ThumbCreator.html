 30: /**
 31:  * Utility to create a thumb.
 32:  * 
 33:  * Please, refer to the `README` file to know how to use the utility and to see examples.
 34:  */
 35: class ThumbCreator {
 36:     /**
 37:      * Height of the origin file
 38:      * @var int
 39:      */
 40:     protected $height;
 41: 
 42:     /**
 43:      * Path of the origin file
 44:      * @var string
 45:      */
 46:     protected $origin;
 47:     
 48:     /**
 49:      * Path of the target file
 50:      * @var string
 51:      */
 52:     protected $target;
 53:     
 54:     /**
 55:      * If the origin file is temporary
 56:      * @var bool 
 57:      */
 58:     protected $temporary = FALSE;
 59: 
 60:     /**
 61:      * Width of the origin file
 62:      * @var int
 63:      */
 64:     protected $width;
 65: 
 66:     /**
 67:      * Construct.  
 68:      * Sets the origin file.
 69:      * 
 70:      * If the path of the origin file is relative, the file will be relative to `APP/webroot/img`.
 71:      * @param string $origin Origin file
 72:      * @return \Thumbs\Utility\ThumbCreator
 73:      * @throws InternalErrorException
 74:      * @uses $height
 75:      * @uses $origin
 76:      * @uses $width
 77:      */
 78:     public function __construct($origin) {
 79:         //Checks for Imagick extension
 80:         if(!extension_loaded('imagick'))
 81:             throw new InternalErrorException(__d('thumbs', '{0} is not available', 'Imagick'));
 82:                 
 83:         //Checks if the target directory is writable
 84:         if(!is_writable(THUMBS))
 85:             throw new InternalErrorException(__d('thumbs', 'File or directory {0} not writeable', THUMBS));
 86:         
 87:         //If the path of the origin file is relative, the file will be relative to `APP/webroot/img`
 88:         if(!Folder::isAbsolute($origin))
 89:             $origin = WWW_ROOT.'img'.DS.$origin;
 90:                 
 91:         //Checks if the origin is an image
 92:         if(!in_array(extension($origin), ['gif', 'jpg', 'jpeg', 'png']))
 93:             throw new InternalErrorException(__d('thumbs', 'The file {0} is not an image', $origin));
 94:         
 95:         //Sets the path, the width and the height of the origin file
 96:         $this->origin = $origin;
 97:         $this->width = getimagesize($origin)[0];
 98:         $this->height = getimagesize($origin)[1];
 99:         
100:         return $this;
101:     }
102:     
103:     /**
104:      * Destruct
105:      * @uses $temporary
106:      * @uses $origin
107:      */
108:     public function __destruct() {
109:         //Removes the origin file, if it's temporary
110:         if($this->temporary)
111:             @unlink($this->origin);
112:     }
113:     
114:     /**
115:      * Downloads a file as a temporary file.  
116:      * This is useful if the origin file is remote.
117:      * @param string $url File url
118:      * @return string Temporary file path
119:      * @throws NotFoundException
120:      * @uses $temporary
121:      */
122:     protected function _downloadTemporary($url) {
123:         $fopen = @fopen($url, 'r');
124:         
125:         //Checks if the file is readable
126:         if(!$fopen)
127:             throw new NotFoundException(__d('thumbs', 'File or directory {0} not readable', $url));
128:         
129:         //Downloads as temporary file
130:         $tmp = sprintf('%s.%s', tempnam(sys_get_temp_dir(), md5($url)), extension($url));
131:         
132:         file_put_contents($tmp, $fopen);
133:         
134:         $this->temporary = TRUE;
135:         
136:         return $tmp;
137:     }
138:     
139:     /**
140:      * Sets and gets the Imagick instance
141:      * @param string $origin Path of the origin file
142:      * @return \Imagick
143:      */
144:     protected function _getImagickInstance($origin) {
145:         //Creates the Imagick instance      
146:         $imagick = new \Imagick($origin);
147:         
148:         //Strips all profiles and comments
149:         $imagick->stripImage();
150:         
151:         //For jpeg images, sets the image compression
152:         if(mime_content_type($origin) === 'image/jpeg') {
153:             $imagick->setImageCompression(\Imagick::COMPRESSION_JPEG);
154:             $imagick->setImageCompressionQuality(100);
155:         }
156:         
157:         return $imagick;
158:     }
159:     
160:     /**
161:      * Creates a thumbnail
162:      * @param int $width Width
163:      * @param int $height Height
164:      * @return string Thumbnail path
165:      * @throws InternalErrorException
166:      * @uses _downloadTemporary()
167:      * @uses _getImagickInstance()
168:      * @uses $height
169:      * @uses $origin
170:      * @uses $temporary
171:      * @uses $width
172:      */
173:     public function resize($width = 0, $height = 0) {       
174:         //Sets the target path
175:         $target = THUMBS.DS.sprintf('resize_%s_w%s_h%s.%s', md5($this->origin), $width, $height, extension($this->origin));
176:         
177:         //If the thumbnail already exists, returns
178:         if(is_readable($target))
179:             return $target;
180:         
181:         //If origin is a remote file, downloads as temporary file
182:         if(is_url($this->origin))
183:             $this->origin = $this->_downloadTemporary($this->origin);
184:         
185:         //Checks if the origin is readable
186:         if(!is_readable($this->origin))
187:             throw new InternalErrorException(__d('thumbs', 'File or directory {0} not readable', $this->origin));
188:         
189:         //If the required size exceed the original size, returns
190:         if(($width && $width >= $this->width) || ($height && $height >= $this->height)) {
191:             //If it's a temporary file, copies as target
192:             if($this->temporary) {
193:                 (new File($this->origin))->copy($target);
194:                 return $target;
195:             }
196:             
197:             return $this->origin;
198:         }
199:         
200:         //Writes the thumbnail
201:         $imagick = $this->_getImagickInstance($this->origin);
202:         $imagick->thumbnailImage($width, $height, $width && $height);
203:         $imagick->writeImage($target);
204:         $imagick->clear();
205:         
206:         return $target;
207:     }
208:     
209:     /**
210:      * Creates a square thumbnail
211:      * @param int $side Side
212:      * @return string Thumbnail path
213:      * @throws InternalErrorException
214:      * @uses _downloadTemporary()
215:      * @uses _getImagickInstance()
216:      * @uses $height
217:      * @uses $origin
218:      * @uses $width
219:      */
220:     public function square($side) {     
221:         //If the required size exceed the original size, so the side is the shortest side
222:         if($side >= $this->width || $side >= $this->height)
223:             $side = $this->width > $this->height ? $this->height : $this->width;
224:         
225:         //Sets the target path
226:         $target = THUMBS.DS.sprintf('square_%s_s%s.%s', md5($this->origin), $side, extension($this->origin));
227:         
228:         //If the thumbnail already exists, returns
229:         if(is_readable($target))
230:             return $target;
231:         
232:         //If origin is a remote file, downloads as temporary file
233:         if(is_url($this->origin))
234:             $this->origin = $this->_downloadTemporary($this->origin);
235:         
236:         //Checks if the origin is readable
237:         if(!is_readable($this->origin))
238:             throw new InternalErrorException(__d('thumbs', 'File or directory {0} not readable', $this->origin));
239:         
240:         //Writes the thumbnail
241:         $imagick = $this->_getImagickInstance($this->origin);
242:         $imagick->cropThumbnailImage($side, $side);
243:         $imagick->writeImage($target);
244:         $imagick->clear();
245:         
246:         return $target;
247:     }
248: }
